6   package shakki.shakki.kayttoliittyma;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.JTextField;
11  import shakki.shakki.logiikka.Logiikka;
12  
13  /**
14   *
15   * @author Sami
16   */
17  public class UusiPeliNappulanKuuntelija implements ActionListener {
18      
19      private Ruutu[][] ruudut;
20      private Logiikka logiikka;
21      private KayttoliittymaTyokalut tyokalut = new KayttoliittymaTyokalut();
22      private JTextField ilmoitusLaatikko;
23  
24      public UusiPeliNappulanKuuntelija(Ruutu[][] ruudut, Logiikka logiikka, JTextField ilmoitusLaatikko) {
25          
26          this.ruudut = ruudut;
27          this.logiikka = logiikka;
28          this.ilmoitusLaatikko = ilmoitusLaatikko;
29          
30      }
31      
32      public void actionPerformed(ActionEvent ae) {
33          
34          tyokalut.asetaRuudutAktiivisiksi(ruudut);
35          logiikka.alustaPelilauta();
36          tyokalut.paivitaPelilauta(ruudut, logiikka.annaPelilauta());
37          
38          ilmoitusLaatikko.setText("Valkoisen vuoro");
39          
40      }
41      
42  }
