19  public class Kayttoliittyma implements Runnable {

31      public void run() {
32          frame = new JFrame("Shakki");
33          frame.setPreferredSize(new Dimension(700, 700));
34          frame.setResizable(false);
35  
36          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
37  
38          luoKomponentit(frame.getContentPane());
39  
40          frame.pack();
41          frame.setVisible(true);
42      }

44      private void luoKomponentit(Container container) {
45  
46          container.setLayout(new BorderLayout());
47          JLabel alusta = new JLabel();
48  
49          alusta.setLayout(new GridLayout(8, 8));
50  
51          JToolBar tools = new JToolBar();
52          tools.setMargin(new Insets(2, 2, 2, 2));
53  
54          JTextField ilmoitusLaatikko = new JTextField("Valkoisen vuoro");
55          ilmoitusLaatikko.setEditable(false);
56          ilmoitusLaatikko.setHorizontalAlignment(JTextField.CENTER);
57          ilmoitusLaatikko.setForeground(Color.blue);
58          
59          //Lisätän uusipeli nappi
60          
61          JButton uusiPeliNappi = new JButton();
62          uusiPeliNappi.addActionListener(new UusiPeliNappulanKuuntelija(ruudut, logiikka, ilmoitusLaatikko));
63          uusiPeliNappi.setText("Uusi Peli");
64          tools.add(uusiPeliNappi);
65          
66          //Lisätään ilmoitusboxi
67          
68          tools.add(ilmoitusLaatikko);
69          
70          container.add(tools, BorderLayout.NORTH);
71          
72          RuutuNappulanKuuntelija kuuntelija = new RuutuNappulanKuuntelija(ruudut, logiikka, ilmoitusLaatikko);
73  
74          //Kuvat
75          ImageIcon sotilasValkoinen = new ImageIcon("src/main/resources/images/sotilas_valkoinen.png");
76          ImageIcon sotilasMusta = new ImageIcon("src/main/resources/images/sotilas_musta.png");
77  
78          ImageIcon hevonenValkoinen = new ImageIcon("src/main/resources/images/hevonen_valkoinen.png");
79          ImageIcon hevonenMusta = new ImageIcon("src/main/resources/images/hevonen_musta.png");
80  
81          ImageIcon torniValkoinen = new ImageIcon("src/main/resources/images/torni_valkoinen.png");
82          ImageIcon torniMusta = new ImageIcon("src/main/resources/images/torni_musta.png");
83  
84          ImageIcon kuningasValkoinen = new ImageIcon("src/main/resources/images/kuningas_valkoinen.png");
85          ImageIcon kuningasMusta = new ImageIcon("src/main/resources/images/kuningas_musta.png");
86  
87          ImageIcon kuningatarValkoinen = new ImageIcon("src/main/resources/images/kuningatar_valkoinen.png");
88          ImageIcon kuningatarMusta = new ImageIcon("src/main/resources/images/kuningatar_musta.png");
89  
90          ImageIcon lahettiValkoinen = new ImageIcon("src/main/resources/images/lahetti_valkoinen.png");
91          ImageIcon lahettiMusta = new ImageIcon("src/main/resources/images/lahetti_musta.png");
92  
93          //1 = musta, 2 = valkonen
94          int ruudunVaritys = 1;
95  
96          for (int i = 0; i <= 7; i++) {
97  
98              for (int j = 0; j <= 7; j++) {
99  
100                 Ruutu nappi = new Ruutu(i, j);
101                 nappi.addActionListener(kuuntelija);
102 
103                 // Lisätään kuvat ruuduille
104                 if (i == 1) {
105 
106                     //nappi.setIcon(sotilasValkoinen);
107                     nappi.setText("\u2659");
108                     nappi.setFont(new Font("", Font.PLAIN, 52));
109 
110                 } else if (i == 6) {
111 
112                     //nappi.setIcon(sotilasMusta);
113                     nappi.setText("\u265F");
114                     nappi.setFont(new Font("", Font.PLAIN, 52));
115 
116                 } else if ((i == 0 && j == 1) || (i == 0 && j == 6)) {
117 
118                     //nappi.setIcon(hevonenValkoinen);
119                     nappi.setText("\u2658");
120                     nappi.setFont(new Font("", Font.PLAIN, 52));
121                     
122                 } else if ((i == 7 && j == 1) || (i == 7 && j == 6)) {
123 
124                     //nappi.setIcon(hevonenMusta);
125                     nappi.setText("\u265E");
126                     nappi.setFont(new Font("", Font.PLAIN, 52));
127 
128                 } else if ((i == 0 && j == 0) || (i == 0 && j == 7)) {
129 
130                     //nappi.setIcon(torniValkoinen);
131                     nappi.setText("\u2656");
132                     nappi.setFont(new Font("", Font.PLAIN, 52));
133 
134                 } else if ((i == 7 && j == 0) || (i == 7 && j == 7)) {
135 
136                     //nappi.setIcon(torniMusta);
137                     nappi.setText("\u265C");
138                     nappi.setFont(new Font("", Font.PLAIN, 52));
139 
140                 } else if (i == 0 && j == 4) {
141 
142                     //nappi.setIcon(kuningasValkoinen);
143                     nappi.setText("\u2654");
144                     nappi.setFont(new Font("", Font.PLAIN, 52));
145 
146                 } else if (i == 7 && j == 4) {
147 
148                     //nappi.setIcon(kuningasMusta);
149                     nappi.setText("\u265A");
150                     nappi.setFont(new Font("", Font.PLAIN, 52));
151 
152                 } else if (i == 0 && j == 3) {
153                                       
154                     //nappi.setIcon(kuningatarValkoinen);
155                     nappi.setText("\u2655");
156                     nappi.setFont(new Font("", Font.PLAIN, 52));
157 
158                 } else if (i == 7 && j == 3) {
159 
160                     //nappi.setIcon(kuningatarMusta);
161                     nappi.setText("\u265B");
162                     nappi.setFont(new Font("", Font.PLAIN, 52));
163 
164                 } else if ((i == 0 && j == 2) || (i == 0 && j == 5)) {
165 
166                     //nappi.setIcon(lahettiValkoinen);
167                     nappi.setText("\u2657");
168                     nappi.setFont(new Font("", Font.PLAIN, 52));
169 
170                 } else if ((i == 7 && j == 2) || (i == 7 && j == 5)) {
171 
172                     //nappi.setIcon(lahettiMusta);
173                     nappi.setText("\u265D");
174                     nappi.setFont(new Font("", Font.PLAIN, 52));
175 
176                 }
177 
178                 if (ruudunVaritys == 1) {
179 
180                     nappi.setBackground(Color.white);
181 
182                 } else {
183 
184                     nappi.setBackground(Color.LIGHT_GRAY);
185                 }
186 
187                 ruudunVaritys = 3 - ruudunVaritys;
188 
189                 ruudut[i][j] = nappi;
190                 alusta.add(ruudut[i][j]);
191 
192             }
193 
194             ruudunVaritys = 3 - ruudunVaritys;
195 
196         }
197 
198         container.add(alusta, BorderLayout.CENTER);
199 
200     }
